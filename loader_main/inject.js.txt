(function() {
    'use strict';

    if (window.__garticUiInjected) return;
    window.__garticUiInjected = true;

    let myWsId = null;
    let activeWs = null;
    const OriginalWebSocket = window.WebSocket;

    // 1. WebSocket Patching & ID Yakalama
    window.WebSocket = function(url, protocols) {
        console.log('[GarticTools] Yeni WebSocket bağlantısı tespit edildi:', url);
        const ws = new OriginalWebSocket(url, protocols);
        
        // Aktif WS'i global olarak tut (Debug için)
        activeWs = ws;
        window._gt_ws = ws; 

        ws.addEventListener('message', function(event) {
            try {
                if (typeof event.data === 'string') {
                    // Yöntem 1: Standart "42["5",..." paketi
                    if (event.data.startsWith('42["5",')) {
                        const jsonStr = event.data.slice(2);
                        const data = JSON.parse(jsonStr);
                        if (Array.isArray(data) && data.length > 2) {
                            myWsId = data[2];
                            window._gt_id = myWsId;
                            updateStatus("Bağlı (ID Alındı)", "#00ff88");
                            console.log('[GarticTools] WebSocket ID yakalandı:', myWsId);
                        }
                    } 
                    // Yöntem 2: Alternatif paket yapısı kontrolü (Güvenlik ağı)
                    else if (event.data.includes('"criador"') || event.data.includes('"idioma"')) {
                         try {
                             const jsonStr = event.data.slice(2);
                             const data = JSON.parse(jsonStr);
                             // Genellikle bu tip büyük paketlerde ID 3. elemandır veya içinde geçer
                             if (Array.isArray(data) && data.length > 2 && typeof data[2] === 'number') {
                                 myWsId = data[2];
                                 window._gt_id = myWsId;
                                 updateStatus("Bağlı (ID Alındı)", "#00ff88");
                                 console.log('[GarticTools] Alternatif yöntemle ID yakalandı:', myWsId);
                             }
                         } catch(e) {}
                    }
                }
            } catch (e) {
                // Hataları sessizce yut
            }
        });

        ws.addEventListener('open', () => {
            console.log('[GarticTools] WebSocket bağlantısı açıldı.');
            updateStatus("Bağlanıyor...", "#ffff00");
        });

        ws.addEventListener('close', () => {
            console.log('[GarticTools] WebSocket bağlantısı kapandı.');
            activeWs = null;
            myWsId = null;
            window._gt_ws = null;
            window._gt_id = null;
            updateStatus("Bağlantı Koptu", "#ff3333");
        });

        ws.addEventListener('error', (err) => {
            console.error('[GarticTools] WebSocket hatası:', err);
            updateStatus("Hata", "#ff3333");
        });

        return ws;
    };

    // Prototype ve statik özellikleri koru
    window.WebSocket.prototype = OriginalWebSocket.prototype;
    window.WebSocket.CONNECTING = OriginalWebSocket.CONNECTING;
    window.WebSocket.OPEN = OriginalWebSocket.OPEN;
    window.WebSocket.CLOSING = OriginalWebSocket.CLOSING;
    window.WebSocket.CLOSED = OriginalWebSocket.CLOSED;

    // 2. UI Oluşturma (Ortalanmış)
    function createUI() {
        const container = document.createElement('div');
        container.id = 'gartic-tools-ui';
        container.style.cssText = `
            position: fixed;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
            width: 320px;
            background: rgba(20, 20, 20, 0.95);
            border: 1px solid #444;
            border-radius: 8px;
            box-shadow: 0 4px 20px rgba(0,0,0,0.6);
            font-family: Arial, sans-serif;
            z-index: 99999;
            color: white;
            display: flex;
            flex-direction: column;
            overflow: hidden;
            transition: all 0.2s ease;
        `;

        // Header
        const header = document.createElement('div');
        header.style.cssText = `
            padding: 12px;
            background: #2a2a2a;
            cursor: move;
            display: flex;
            justify-content: space-between;
            align-items: center;
            border-bottom: 1px solid #444;
            user-select: none;
        `;
        header.innerHTML = `<span style="font-weight:bold; font-size:14px;">🎨 Gartic Chat</span><span id="gt-status" style="font-size:12px; color:#aaa;">Bekleniyor...</span>`;
        
        // Content
        const content = document.createElement('div');
        content.style.cssText = `padding: 15px;`;

        // Input Area
        const inputGroup = document.createElement('div');
        inputGroup.style.cssText = `display: flex; gap: 8px; margin-top: 5px;`;

        const input = document.createElement('input');
        input.type = 'text';
        input.placeholder = 'Mesajınızı yazın...';
        input.style.cssText = `
            flex: 1;
            padding: 10px;
            border: 1px solid #555;
            border-radius: 4px;
            background: #1a1a1a;
            color: white;
            outline: none;
            font-size: 14px;
        `;

        const btn = document.createElement('button');
        btn.innerText = 'Gönder';
        btn.style.cssText = `
            padding: 10px 15px;
            background: #007bff;
            border: none;
            border-radius: 4px;
            color: white;
            cursor: pointer;
            font-weight: bold;
            font-size: 14px;
            transition: background 0.2s;
        `;
        btn.onmouseover = () => btn.style.background = '#0056b3';
        btn.onmouseout = () => btn.style.background = '#007bff';

        inputGroup.appendChild(input);
        inputGroup.appendChild(btn);
        content.appendChild(inputGroup);
        container.appendChild(header);
        container.appendChild(content);
        document.body.appendChild(container);

        // Sürükleme Mantığı
        let isDragging = false;
        let offsetX, offsetY;

        header.addEventListener('mousedown', (e) => {
            isDragging = true;
            const rect = container.getBoundingClientRect();
            offsetX = e.clientX - rect.left;
            offsetY = e.clientY - rect.top;
            container.style.transform = 'none';
            container.style.top = rect.top + 'px';
            container.style.left = rect.left + 'px';
        });

        document.addEventListener('mousemove', (e) => {
            if (isDragging) {
                container.style.left = `${e.clientX - offsetX}px`;
                container.style.top = `${e.clientY - offsetY}px`;
            }
        });

        document.addEventListener('mouseup', () => {
            isDragging = false;
        });

        // Gönderme Mantığı
        const sendMessage = () => {
            const msg = input.value.trim();
            if (!msg) return;
            
            if (!myWsId) {
                alert('Hata: WebSocket ID henüz alınmadı. Lütfen odaya girin ve bekleyin.');
                console.warn('[GarticTools] myWsId boş. Mevcut ID:', window._gt_id);
                return;
            }
            if (!activeWs || activeWs.readyState !== WebSocket.OPEN) {
                alert('Hata: Aktif WebSocket bağlantısı bulunamadı veya açık değil.');
                return;
            }

            const packet = `42["11","${myWsId}","${msg}"]`;
            console.log('[GarticTools] Paket gönderiliyor:', packet);
            activeWs.send(packet);
            input.value = ''; 
            input.focus();
        };

        btn.addEventListener('click', sendMessage);
        input.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendMessage();
        });

        // Status Güncelleme Fonksiyonu
        window.updateStatus = (text, color) => {
            const statusEl = document.getElementById('gt-status');
            if (statusEl) {
                statusEl.innerText = text;
                statusEl.style.color = color;
            }
        };
    }

    // Sayfa yüklendiğinde UI'ı oluştur
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', createUI);
    } else {
        createUI();
    }

})();
