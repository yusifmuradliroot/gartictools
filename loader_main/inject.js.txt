(function() {
    'use strict';

    if (window.__stealthLoggerInjected) return;
    window.__stealthLoggerInjected = true;

    let myWebSocketId = null;
    const originalWebSocket = window.WebSocket;

    // WebSocket sınıfını sarmalıyoruz
    window.WebSocket = class extends originalWebSocket {
        constructor(url) {
            super(url);

            // Websocket bağlantısı kurulduğunda gelen "init" mesajını dinle
            this.addEventListener('message', function(event) {
                try {
                    // Sadece string mesajları işle
                    if (typeof event.data === 'string') {
                        // Mesajın "42[\"5\",..." ile başlayıp başlamadığını kontrol et
                        if (event.data.startsWith('42["5",')) {
                            // JSON.parse ile ayrıştır
                            const parts = JSON.parse(event.data.replace(/^42/, ''));
                            // Üçüncü parametre (index 2) mywebsocketid'dir
                            if (parts && parts.length > 2) {
                                myWebSocketId = parts[2];
                                console.log('[StealthLogger] mywebsocketid bulundu:', myWebSocketId);
                            }
                        }
                    }
                } catch (e) {
                    // Hatalar sessizce göz ardı edilir
                }
            });

            // Orijinal send fonksiyonunu sakla
            const originalSend = this.send;
            this.send = function(data) {
                return originalSend.apply(this, arguments);
            };
        }
    };

    // Konsol komutunu tanımla: garticchat(send("mesaj"))
    window.garticchat = function(callback) {
        if (typeof callback !== 'function') {
            console.error('Kullanım: garticchat(send("mesajınız"))');
            return;
        }

        if (!myWebSocketId) {
            console.error('Hata: mywebsocketid henüz alınmadı. Lütfen odaya bağlanın.');
            return;
        }

        // Callback'ten gerçek mesajı al (örneğin: send("merhaba"))
        const message = callback();
        if (!message || typeof message !== 'string') {
            console.error('Geçersiz mesaj.');
            return;
        }

        // Özel formatlı mesajı oluştur
        const packet = `42["11","${myWebSocketId}","${message}"]`;
        
        // Tüm açık WebSocket bağlantılarında dene
        for (const ws of window.__wsInstances || []) {
            if (ws.readyState === WebSocket.OPEN) {
                ws.send(packet);
                console.log(`[GarticChat] Mesaj gönderildi: ${message}`);
                break;
            }
        }
    };

    // Kolaylık için send fonksiyonunu dışa aktar
    window.send = function(text) { return text; };

    // Açık WebSocket örneklerini takip et (send için)
    const instances = [];
    const originalWS = window.WebSocket;
    window.WebSocket = function(...args) {
        const instance = new originalWS(...args);
        instances.push(instance);
        instance.addEventListener('close', () => {
            const index = instances.indexOf(instance);
            if (index !== -1) instances.splice(index, 1);
        });
        return instance;
    };
    window.__wsInstances = instances;
    window.WebSocket.prototype = originalWS.prototype;
    Object.assign(window.WebSocket, originalWS);

})();
