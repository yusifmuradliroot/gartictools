(function() {
    'use strict';

    if (window.__garticUiInjected) return;
    window.__garticUiInjected = true;

    let myWsId = null;
    let activeWs = null;
    const OriginalWebSocket = window.WebSocket;

    // 1. WebSocket Patching & ID Yakalama
    window.WebSocket = function(url, protocols) {
        const ws = new OriginalWebSocket(url, protocols);
        activeWs = ws;

        ws.addEventListener('message', function(event) {
            try {
                if (typeof event.data === 'string' && event.data.startsWith('42["5",')) {
                    const jsonStr = event.data.slice(2);
                    const data = JSON.parse(jsonStr);
                    if (Array.isArray(data) && data.length > 2) {
                        myWsId = data[2];
                        updateStatus("Bağlı", "#00ff88");
                    }
                }
            } catch (e) {}
        });

        ws.addEventListener('close', () => {
            activeWs = null;
            updateStatus("Bağlantı Koptu", "#ff3333");
        });

        return ws;
    };

    window.WebSocket.prototype = OriginalWebSocket.prototype;
    window.WebSocket.CONNECTING = OriginalWebSocket.CONNECTING;
    window.WebSocket.OPEN = OriginalWebSocket.OPEN;
    window.WebSocket.CLOSING = OriginalWebSocket.CLOSING;
    window.WebSocket.CLOSED = OriginalWebSocket.CLOSED;

    // 2. UI Oluşturma
    function createUI() {
        const container = document.createElement('div');
        container.id = 'gartic-tools-ui';
        container.style.cssText = `
            position: fixed;
            bottom: 20px;
            right: 20px;
            width: 300px;
            background: rgba(20, 20, 20, 0.95);
            border: 1px solid #444;
            border-radius: 8px;
            box-shadow: 0 4px 15px rgba(0,0,0,0.5);
            font-family: Arial, sans-serif;
            z-index: 99999;
            color: white;
            display: flex;
            flex-direction: column;
            overflow: hidden;
            transition: all 0.3s ease;
        `;

        // Header (Sürükleme Alanı)
        const header = document.createElement('div');
        header.style.cssText = `
            padding: 10px;
            background: #333;
            cursor: move;
            display: flex;
            justify-content: space-between;
            align-items: center;
            border-bottom: 1px solid #444;
        `;
        header.innerHTML = `<span style="font-weight:bold; font-size:14px;">🎨 Gartic Tools</span><span id="gt-status" style="font-size:12px; color:#aaa;">Bekleniyor...</span>`;
        
        // Content
        const content = document.createElement('div');
        content.style.cssText = `padding: 10px;`;

        // Input Area
        const inputGroup = document.createElement('div');
        inputGroup.style.cssText = `display: flex; gap: 5px; margin-top: 5px;`;

        const input = document.createElement('input');
        input.type = 'text';
        input.placeholder = 'Mesajınızı yazın...';
        input.style.cssText = `
            flex: 1;
            padding: 8px;
            border: 1px solid #555;
            border-radius: 4px;
            background: #222;
            color: white;
            outline: none;
        `;

        const btn = document.createElement('button');
        btn.innerText = 'Gönder';
        btn.style.cssText = `
            padding: 8px 12px;
            background: #007bff;
            border: none;
            border-radius: 4px;
            color: white;
            cursor: pointer;
            font-weight: bold;
        `;
        btn.onmouseover = () => btn.style.background = '#0056b3';
        btn.onmouseout = () => btn.style.background = '#007bff';

        inputGroup.appendChild(input);
        inputGroup.appendChild(btn);
        content.appendChild(inputGroup);
        container.appendChild(header);
        container.appendChild(content);
        document.body.appendChild(container);

        // Sürükleme Mantığı
        let isDragging = false;
        let offsetX, offsetY;

        header.addEventListener('mousedown', (e) => {
            isDragging = true;
            offsetX = e.clientX - container.getBoundingClientRect().left;
            offsetY = e.clientY - container.getBoundingClientRect().top;
        });

        document.addEventListener('mousemove', (e) => {
            if (isDragging) {
                container.style.left = `${e.clientX - offsetX}px`;
                container.style.top = `${e.clientY - offsetY}px`;
                container.style.bottom = 'auto';
                container.style.right = 'auto';
            }
        });

        document.addEventListener('mouseup', () => {
            isDragging = false;
        });

        // Gönderme Mantığı
        const sendMessage = () => {
            const msg = input.value.trim();
            if (!msg) return;
            
            if (!myWsId) {
                alert('Hata: WebSocket ID henüz alınmadı. Lütfen odaya girin.');
                return;
            }
            if (!activeWs || activeWs.readyState !== WebSocket.OPEN) {
                alert('Hata: Aktif bağlantı yok.');
                return;
            }

            const packet = `42["11","${myWsId}","${msg}"]`;
            activeWs.send(packet);
            input.value = ''; // Kutuyu temizle
            input.focus();
        };

        btn.addEventListener('click', sendMessage);
        input.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendMessage();
        });

        // Status Güncelleme Fonksiyonu
        window.updateStatus = (text, color) => {
            const statusEl = document.getElementById('gt-status');
            if (statusEl) {
                statusEl.innerText = text;
                statusEl.style.color = color;
            }
        };
    }

    // Sayfa yüklendiğinde UI'ı oluştur
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', createUI);
    } else {
        createUI();
    }

    function garticchat(message) {
        if (typeof message !== 'string') {
            console.error('[GarticChat] Hata: Mesaj bir metin (string) olmalıdır.');
            return;
        }
        if (!myWsId) {
            console.error('[GarticChat] Hata: WebSocket ID henüz alınmadı. Lütfen odaya girin.');

        if (!activeWs || activeWs.readyState !== WebSocket.OPEN) {
            console.error('[GarticChat] Hata: Aktif WebSocket bağlantısı bulunamadı.');
            return;
        }

        const packet = `42["11","${myWsId}","${message}"]`;
        activeWs.send(packet);
        console.log(`[GarticChat] Gönderildi: ${message}`);
    }
