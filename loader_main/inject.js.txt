(function() {
    'use strict';

    if (window.__GarticSenderInjected) return;
    window.__GarticSenderInjected = true;

    let myWebSocketId = null;
    let activeWsInstance = null;
    const originalWebSocket = window.WebSocket;

    // WebSocket sınıfını patch'le
    window.WebSocket = class extends originalWebSocket {
        constructor(url) {
            super(url);
            activeWsInstance = this; // En son açılan bağlantıyı takip et

            // ID yakalama
            this.addEventListener('message', function(event) {
                try {
                    if (typeof event.data === 'string' && event.data.startsWith('42["5",')) {
                        const jsonStr = event.data.substring(2);
                        const parts = JSON.parse(jsonStr);
                        if (parts && parts.length > 2) {
                            myWebSocketId = parts[2];
                            console.log('[GarticTools] mywebsocketid alındı:', myWebSocketId);
                        }
                    }
                } catch (e) {}
            });

            // Orijinal send'i koru
            const originalSend = this.send;
            this.send = function(data) {
                return originalSend.apply(this, arguments);
            };
        }
    };

    window.WebSocket.prototype = originalWebSocket.prototype;
    Object.assign(window.WebSocket, originalWebSocket);

    // Mesaj gönderme fonksiyonu
    const sendFunction = function(message) {
        if (!myWebSocketId) {
            console.error('[GarticTools HATA] mywebsocketid henüz alınmadı. Lütfen odaya bağlanın.');
            return;
        }
        if (typeof message !== 'string') {
            console.error('[GarticTools HATA] Mesaj bir metin olmalıdır.');
            return;
        }
        if (!activeWsInstance || activeWsInstance.readyState !== WebSocket.OPEN) {
            console.error('[GarticTools HATA] Aktif WebSocket bağlantısı bulunamadı.');
            return;
        }

        const packet = `42["11","${myWebSocketId}","${message}"]`;
        activeWsInstance.send(packet);
        console.log(`[GarticTools] Gönderildi: ${message}`);
    };

    // 🔥 GLOBAL KAPSAMA ZORLA EKLE
    window.gsend = sendFunction;
    if (typeof unsafeWindow !== 'undefined') {
        unsafeWindow.gsend = sendFunction;
    }

    // Doğrulama logu
    console.log('%c[GarticTools] Hazır! Konsola gsend("mesaj") yazabilirsiniz.', 'color: #00ff88; font-weight: bold;');
    console.log('[GarticTools] gsend tanımlı mı?', typeof window.gsend);
})();
