(function() {
    'use strict';

    if (window.__garticChatInjected) return;
    window.__garticChatInjected = true;

    let myWsId = null;
    let activeWs = null;
    const OriginalWebSocket = window.WebSocket;

    // WebSocket constructor'ını güvenli şekilde patch'le
    window.WebSocket = function(url, protocols) {
        const ws = new OriginalWebSocket(url, protocols);
        activeWs = ws;

        ws.addEventListener('message', function(event) {
            try {
                if (typeof event.data === 'string' && event.data.startsWith('42["5",')) {
                    const jsonStr = event.data.slice(2);
                    const data = JSON.parse(jsonStr);
                    if (Array.isArray(data) && data.length > 2) {
                        myWsId = data[2];
                        console.log('[GarticChat] WebSocket ID yakalandı:', myWsId);
                    }
                }
            } catch (e) {}
        });

        return ws;
    };

    // Prototype ve statik özellikleri koru
    window.WebSocket.prototype = OriginalWebSocket.prototype;
    window.WebSocket.CONNECTING = OriginalWebSocket.CONNECTING;
    window.WebSocket.OPEN = OriginalWebSocket.OPEN;
    window.WebSocket.CLOSING = OriginalWebSocket.CLOSING;
    window.WebSocket.CLOSED = OriginalWebSocket.CLOSED;

    // Ana fonksiyon
    function garticchat(message) {
        if (typeof message !== 'string') {
            console.error('[GarticChat] Hata: Mesaj bir metin (string) olmalıdır.');
            return;
        }
        if (!myWsId) {
            console.error('[GarticChat] Hata: WebSocket ID henüz alınmadı. Lütfen odaya girin.');
            return;
        }
        if (!activeWs || activeWs.readyState !== WebSocket.OPEN) {
            console.error('[GarticChat] Hata: Aktif WebSocket bağlantısı bulunamadı.');
            return;
        }

        const packet = `42["11","${myWsId}","${message}"]`;
        activeWs.send(packet);
        console.log(`[GarticChat] Gönderildi: ${message}`);
    }

    // Global pencereye doğrudan bağla
    window.garticchat = garticchat;
    if (typeof unsafeWindow !== 'undefined') {
        unsafeWindow.garticchat = garticchat;
    }

    console.log('%c[GarticChat] Hazır! Konsola garticchat("mesaj") yazabilirsiniz.', 'color: #00ff88; font-weight: bold;');
})();
