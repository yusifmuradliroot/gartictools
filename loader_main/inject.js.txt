(function() {
    'use strict';

    if (window.__GarticSenderInjected) return;
    window.__GarticSenderInjected = true;

    let myWebSocketId = null;
    const originalWebSocket = window.WebSocket;

    // WebSocket sınıfını sarmala
    window.WebSocket = class extends originalWebSocket {
        constructor(url) {
            super(url);

            // "42["5",...]" mesajından mywebsocketid'yi al
            this.addEventListener('message', function(event) {
                try {
                    if (typeof event.data === 'string' && event.data.startsWith('42["5",')) {
                        const parts = JSON.parse(event.data.replace(/^42/, ''));
                        if (parts.length > 2) {
                            myWebSocketId = parts[2];
                        }
                    }
                } catch (e) {
                    // Hata sessizce göz ardı edilir
                }
            });

            // Orijinal send fonksiyonunu sakla
            const originalSend = this.send;
            this.send = function(data) {
                return originalSend.apply(this, arguments);
            };
        }
    };

    // Prototip zincirini koru
    window.WebSocket.prototype = originalWebSocket.prototype;
    Object.assign(window.WebSocket, originalWebSocket);

    // GLOBAL: gsend fonksiyonunu tanımla
    window.gsend = function(message) {
        if (!myWebSocketId) {
            console.error('[HATA] mywebsocketid alınamadı. Lütfen odaya bağlanın.');
            return;
        }

        if (typeof message !== 'string') {
            console.error('[HATA] Mesaj bir metin olmalıdır.');
            return;
        }

        // Özel formatlı paketi oluştur
        const packet = `42["11","${myWebSocketId}","${message}"]`;

        // Tüm açık WS bağlantılarında dene
        for (const ws of window.__wsInstances || []) {
            if (ws.readyState === WebSocket.OPEN) {
                ws.send(packet);
                break;
            }
        }
    };

    // Açık WS bağlantılarını takip et
    const instances = [];
    const originalWS = window.WebSocket;
    window.WebSocket = function(...args) {
        const instance = new originalWS(...args);
        instances.push(instance);
        instance.addEventListener('close', () => {
            const index = instances.indexOf(instance);
            if (index !== -1) instances.splice(index, 1);
        });
        return instance;
    };
    window.__wsInstances = instances;
    window.WebSocket.prototype = originalWS.prototype;
    Object.assign(window.WebSocket, originalWS);

    console.log('%c[gsend] Hazır. Konsola gsend("merhaba") yazarak mesaj gönderebilirsin.', 'color: #00ff88;');
})();
